\section{Functions}

\newcommand{\Jand}{,\,}
\newcommand{\JfromEnv}{\Gamma \vdash}
\newcommand{\Jis}[2]{{#1}:\,{#2}}
\newcommand{\Jdef}[2]{{#1}\;\equiv\;{#2}}

\newcommand{\Htype}{\texttt{Type}}
\newcommand{\HfnTy}[2]{\texttt{(${#1}$)\;->\;${#2}$}}
\newcommand{\HfnVal}[3]{\texttt{(${#1}$)\;->\;${#2}$\;=>\;${#3}$}}
\newcommand{\HfnCall}[2]{\texttt{${#1}$(${#2}$)}}
\newcommand{\Hparam}[2]{\texttt{{#1}:\,${#2}$}}
\newcommand{\Hcomma}{\texttt{,\;}}

\begin{center}
  \begin{prooftree}
    \hypo{\JfromEnv & \Jis{A, B, \ldots}{\Htype} \Jand \Jis{R}{\Htype}}
    \infer1[Function type introduction]{
      \JfromEnv &
      \Jis{
        \HfnTy{\Hparam{a}{A} \Hcomma \Hparam{b}{B} \Hcomma \ldots}{R}
      }{\Htype}
    }
  \end{prooftree}
\end{center}

\begin{center}
  \begin{prooftree}
    \hypo{\JfromEnv & \Jis{A, B, \ldots}{\Htype} \Jand \Jis{R}{\Htype} \Jand \Jis{r}{R}}
    \infer1[Function value introduction]{
      \JfromEnv &
      \Jis{
        \HfnVal{\Hparam{a}{A} \Hcomma \Hparam{b}{B} \Hcomma \ldots}{R}{r}
      }{
        \HfnTy{\Hparam{a}{A} \Hcomma \Hparam{b}{B} \Hcomma \ldots}{R}
      }
    }
  \end{prooftree}
\end{center}

\begin{center}
  \begin{prooftree}
    \hypo{\JfromEnv &
      \begin{matrix*}[l]
        \Jis{A, B, \ldots}{\Htype} \Jand \\
        \Jis{R}{\Htype} \Jand            \\
        \Jis{f}{
          \HfnTy{\Hparam{a}{A} \Hcomma \Hparam{b}{B} \Hcomma \ldots}{R}
        } \Jand                          \\
        \Jis{x}{A} \Jand \Jis{y}{B} \Jand \ldots
      \end{matrix*}
    }
    \infer1[Function elimination]{\JfromEnv & \Jis{\HfnCall{f}{x \Hcomma y \Hcomma \ldots}}{R}}
  \end{prooftree}
\end{center}
